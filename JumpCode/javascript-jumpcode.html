📦 Variables & Data Types

var, let, const
Declaring variables with different scopes
var name = "John"; // Function scoped, can be redeclared let age = 25; // Block scoped, can be reassigned const email = "john@example.com"; // Block scoped, cannot be reassigned
Primitive Data Types
Basic data types in JavaScript
let str = "Hello"; // String let num = 42; // Number let bool = true; // Boolean let nothing = null; // Null let undef = undefined; // Undefined let sym = Symbol("id"); // Symbol let big = 123n; // BigInt
Objects & Arrays
Complex data structures
let person = { name: "John", age: 25, email: "john@example.com" }; let colors = ["red", "blue", "green"]; let mixed = [1, "two", true, {name: "obj"}];

🔧 Operators

Arithmetic Operators
Basic math operations
let a = 10, b = 3; console.log(a + b); // 13 (Addition) console.log(a - b); // 7 (Subtraction) console.log(a * b); // 30 (Multiplication) console.log(a / b); // 3.33 (Division) console.log(a % b); // 1 (Modulo) console.log(a ** b); // 1000 (Exponentiation)
Comparison Operators
Comparing values
let a = 5, b = "5"; console.log(a == b); // true (Equal value) console.log(a === b); // false (Equal value AND type) console.log(a != b); // false (Not equal) console.log(a !== b); // true (Not equal or type) console.log(a > 3); // true (Greater than) console.log(a < 10); // true (Less than)
Logical Operators
Boolean logic
let a = true, b = false; console.log(a && b); // false (AND) console.log(a || b); // true (OR) console.log(!a); // false (NOT) if (a && !b) { console.log("Both conditions are true!"); }
Assignment Operators
Assigning and updating values
let x = 10; x += 5; // x = x + 5 (15) x -= 3; // x = x - 3 (12) x *= 2; // x = x * 2 (24) x /= 4; // x = x / 4 (6) x %= 2; // x = x % 2 (0) console.log(x);

🎯 Control Flow

if, else if, else
Conditional statements
let age = 18; if (age < 13) { console.log("Child"); } else if (age < 18) { console.log("Teenager"); } else { console.log("Adult"); }
switch Statement
Multiple condition branches
let day = 3; let dayName; switch(day) { case 1: dayName = "Monday"; break; case 2: dayName = "Tuesday"; break; case 3: dayName = "Wednesday"; break; default: dayName = "Unknown"; }
Loops: for, while, do...while
Repeating code blocks
// For loop for (let i = 0; i < 5; i++) { console.log(i); } // While loop let count = 0; while (count < 5) { console.log(count); count++; } // Do...while (runs at least once) let x = 0; do { console.log(x); x++; } while (x < 5);
for...in & for...of
Iterating over collections
let obj = {name: "John", age: 25}; let arr = ["red", "blue", "green"]; // for...in (iterates over keys/indices) for (let key in obj) { console.log(key, obj[key]); } // for...of (iterates over values) for (let color of arr) { console.log(color); }

⚙️ Functions

Function Declaration
Traditional function definition
function greet(name) { return "Hello, " + name + "!"; } console.log(greet("John")); // "Hello, John!"
Arrow Functions
Modern function syntax
const add = (a, b) => a + b; const greet = name => `Hello, ${name}!`; const getRandom = () => Math.random(); console.log(add(5, 3)); // 8 console.log(greet("Alice")); // Hello, Alice!
Default Parameters & Rest Parameters
Function parameters with defaults and variadic functions
function greet(name = "Guest") { console.log("Hello, " + name); } function sum(...numbers) { return numbers.reduce((a, b) => a + b, 0); } greet(); // "Hello, Guest" console.log(sum(1, 2, 3, 4, 5)); // 15
Higher-Order Functions
Functions that work with other functions
// Function as parameter function map(arr, callback) { return arr.map(callback); } // Function as return value function makeMultiplier(n) { return (x) => x * n; } const double = makeMultiplier(2); console.log(double(5)); // 10

📋 Array Methods

Common Array Methods
Essential array operations
let arr = [1, 2, 3, 4, 5]; // map() - Transform each element let doubled = arr.map(x => x * 2); // [2, 4, 6, 8, 10] // filter() - Keep elements that pass test let evens = arr.filter(x => x % 2 === 0); // [2, 4] // reduce() - Combine into single value let sum = arr.reduce((a, b) => a + b, 0); // 15 // find() - Get first matching element let first = arr.find(x => x > 3); // 4 // includes() - Check if value exists let has3 = arr.includes(3); // true // sort() - Sort array arr.sort((a, b) => a - b);
Adding & Removing Elements
Modifying array contents
let arr = [1, 2, 3]; arr.push(4); // Add at end: [1, 2, 3, 4] arr.pop(); // Remove from end: [1, 2, 3] arr.unshift(0); // Add at start: [0, 1, 2, 3] arr.shift(); // Remove from start: [1, 2, 3] arr.splice(1, 1, 10); // Remove 1 at index 1, insert 10: [1, 10, 3]

📝 String Methods

String Manipulation
Common string operations
let str = "Hello World"; str.length; // 11 str.toUpperCase(); // "HELLO WORLD" str.toLowerCase(); // "hello world" str.charAt(0); // "H" str.indexOf("World"); // 6 str.substring(0, 5); // "Hello" str.split(" "); // ["Hello", "World"] str.replace("World", "JS"); // "Hello JS" str.trim(); // Remove whitespace str.startsWith("Hello"); // true str.endsWith("World"); // true str.includes("lo"); // true str.repeat(2); // "Hello WorldHello World"

String interpolation with backticks
let name = "John"; let age = 25; let message = `Hello, ${name}! You are ${age} years old.`; // "Hello, John! You are 25 years old." let multiline = ` Line 1 Line 2 Line 3 `;

🏢 Objects & Classes

Object Literals
Creating objects
let person = { name: "John", age: 25, email: "john@example.com", greet: function() { return "Hello, " + this.name; } }; console.log(person.name); // "John" console.log(person.greet()); // "Hello, John"
Classes
Object-oriented programming
class Person { constructor(name, age) { this.name = name; this.age = age; } greet() { return `Hello, I'm ${this.name} and I'm ${this.age} years old.`; } haveBirthday() { this.age++; } } let person = new Person("John", 25); console.log(person.greet()); // "Hello, I'm John and I'm 25 years old."
Object Methods & Destructuring
Useful object operations
let obj = {name: "John", age: 25, city: "NYC"}; Object.keys(obj); // ["name", "age", "city"] Object.values(obj); // ["John", 25, "NYC"] Object.entries(obj); // [["name","John"], ["age",25], ["city","NYC"]] // Destructuring const {name, age} = obj; console.log(name, age); // "John" 25 const [a, b, c] = [1, 2, 3]; console.log(a); // 1

⏱️ Asynchronous Programming

setTimeout & setInterval
Delayed and repeated execution
// Execute after 2 seconds setTimeout(() => { console.log("After 2 seconds"); }, 2000); // Execute every 1 second (10 times) let count = 0; let interval = setInterval(() => { count++; console.log("Count: " + count); if (count === 10) clearInterval(interval); }, 1000);
Promises
Handle asynchronous operations
let promise = new Promise((resolve, reject) => { setTimeout(() => { resolve("Success!"); }, 2000); }); promise .then(result => console.log(result)) .catch(error => console.log(error)); // Promise.all() - Wait for all promises Promise.all([promise1, promise2]) .then(results => console.log(results));
async & await
Modern async syntax
async function fetchData() { try { let response = await fetch("https://api.example.com/data"); let data = await response.json(); console.log(data); } catch (error) { console.log("Error: " + error); } } fetchData();

💡 Pro Tips:
  • Use const by default, let when needed, avoid var
  • Arrow functions are concise but don't have their own this context
  • Always use strict equality === instead of ==
  • Use async/await instead of .then() for better readability
  • Always handle errors in async operations with try/catch
  • Use template literals for cleaner string concatenation
  • Remember that array and object methods are immutable - they don't modify the original
